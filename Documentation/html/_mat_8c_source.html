00571 //----------------------------------------------------------
00572 int MatPseudoInv( Reel *_inv , const Reel *_mat , int _m , int _n )
00573 {
00574     /*
00575     Les noms des vecteurs de l'algorithme de Greville (d,b,c) ont été conservé
00576     Biblio : Théorie Calcul Matriciel p 129 (bibli Lirmm n° 7068)
00577 
00578     _inv est calculé ligne par ligne
00579 
00580     indice k = [0 ŕ n[
00581 
00582     A   : matrice [m,n]
00583     A+  : pseudo [n,m]
00584     ak  : colonne k de A [m,1]
00585     akT : transposée de ak ou ligne k de AT [1,m]
00586     Ak  : matrice formée des k premičres colonnes de A [m,k-1]
00587     Ak+ : matrice formée des k premičres lignes de A+ [k-1,m]
00588 
00589      !!! faire + rapide
00590      !!! si _m < _n utiliser (A+)t = (At)+ si + rapide ŕ mesurer sous RTX
00591      !!! voir en récursif Non car le pas est complexe par rapport ŕ l'ensemble de l'algorithme
00592                               męme si le test d'arręt et le premier pas sont simples.
00593      !!! voir pour k = 0 si on l'intégre dans la boucle
00594 
00595 
00596     Pour mémoire :
00597     - Pseudo inverse ŕ droite : Pinv(M)=Mt*(M*Mt)^-1
00598     - Pseudo inverse ŕ gauche : Pinv(M)=(Mt*M)^-1*Mt
00599     */
00600 
00601 #define Pr 0 // print dans la boucle de calcul
00602 
00603 #define EPS (Reel) 1e-16 // -12 chez Xavier
00604 
00605     int  k ;
00606     Reel tmat[NMax * MMax] , // transposée de _mat(_n,_m)
00607     *pat ,              // pointe la ligne k de tmat (pour la colonne k de _mat)
00608     xnorm ,
00609     xnul ,  // groupé avec xmax
00610     d[NMax-1] ,    // vecteur colonne de taille k variable, maximum _n-1, utilisé aussi pour td (transposée de d)
00611     x ,
00612     matAux[( NMax - 1 ) * MMax] ,
00613     *ctPuisb ; // pointeur sur un vecteur ligne ct puis b , pointe la ligne k de _inv
00614 
00615 //------------------
00616     if ( ( _m > MMax ) || ( _n > NMax ) )
00617     {
00618         printf("\nMatPseudoInv: erreur matrice %2d x %2d maximum.", MMax , NMax ) ; // !!! message err
00619         return 1 ;
00620     }
00621 
00622     MatTrans( tmat , _mat , _m , _n ) ; // tmat(_n,_m) = transposée de _mat l'accé aux lignes y est plus pratique
00623 
00624     for ( xnul = (Reel) 0.f , k = _n ; k-- ; ) // sur les _n vecteurs lignes de tmat
00625     {
00626         xnorm = NormeCarre( tmat + k * _m , _m ) ; // ligne k de tmat, tmat ŕ _m colonnes
00627         if ( xnorm > xnul ) xnul = xnorm ; // garder la plus grande
00628     }
00629 // Le dernier xnorm calculé est celui de la ligne 0 de tmat, soit la colonne 0 de _mat
00630 
00631     if ( xnul > (Reel) 1e32 ) return 2 ; // erreur, n'est pas dans Xavier
00632     xnul *= EPS ;
00633 
00634     pat     = tmat ; // on utilise les lignes de tmat plus facile d'accčs que les colonnes de _mat
00635     ctPuisb = _inv ; // on utilise la premičre ligne non encore calculée de _inv pour ranger ck puis bk
00636 
00637 
00638 //si a0 = 0 alors A1+ = 0                    // ligne 0 seule de la speudo [1,m]
00639 //          sinon A1+ = (a0T * a0)^-1 * a0T  // ( 1 / ( [1,m] * [m,1] ) ) * [1,m] = [1,m]
00640 
00641 // Si la premičre colonne de _mat n'est pas nulle
00642     if ( xnorm > xnul ) MatMulScal( ctPuisb , (Reel) 1.f / xnorm , pat , 1 , _m ) ; // init 1čre ligne de _inv k = 0
00643     else                MatClr( ctPuisb , 1 , _m ) ; // clear 1čre ligne de _inv si la norme est nulle
00644 
00645 #if Pr
00646     MatPrint( "_inv" , " %10.6f" , _inv , 1 , _m ) ; // k = 0
00647 #endif
00648 
00649     for ( k = 1 ; k < _n ; k++ )
00650     {
00651         pat     += _m ; // passe ŕ la ligne suivante, pointe sur la ligne k de tmat (0 ŕ n-1) colonne k de _mat
00652         ctPuisb += _m ; // passe ŕ la ligne suivante, pointe sur la ligne k de _inv
00653 
00654         // d = Ak+ * ak     // [k,m] * [m,1] = [k,1] ce vecteur auxiliaire peut ętre écrasé ŕ chaque boucle donc pas d'indice pour d
00655         MatMul( d , _inv , pat , k , _m , 1 ) ; // at est identique ŕ a
00656         // ck = ak - Ak * d // [m,1] - [m,k] * [k,1] = [m,1] sera rangé provisoirement dans la ligne k de la pseudo, elle est libre
00657         // Ak étant difficilement accessible on utilise la relation : _mat * d = t(td * tmat) ; [m,k] * [k,1] = t([1,k] * [k,m]) = [m,1]
00658         MatMul( ctPuisb , d , tmat , 1 , k , _m ) ;  // utilise d pour transposé de d
00659         MatSub( ctPuisb , pat , ctPuisb , 1 , _m ) ; // ligne k de tmat et on utilise la place de _inv
00660 
00661         //si ck = 0 alors bk = (1 + dT * d)^-1 * dT * Ak+ // scalaire * [1,k] * [k,m] = [1,m]
00662         //          sinon bk = (ckT * ck)^-1 * ckT  // ( 1 / ( [1,m] * [m,1] ) ) * [1,m] = [1,m]
00663 
00664         xnorm = NormeCarre( ctPuisb , _m ) ;
00665         if ( xnorm > xnul ) MatMulScal( ctPuisb , (Reel) 1.f / xnorm , ctPuisb , 1 , _m ) ; // c'est bk
00666         else             // Si cette norme est nulle
00667         {
00668             x = (Reel) 1.f / ( (Reel) 1.f + NormeCarre( d , k ) ) ;
00669             MatMul( ctPuisb , d , _inv , 1 , k , _m ) ;    // cPuisb = td * _inv (les k 1čres lignes)
00670             MatMulScal( ctPuisb , x , ctPuisb , 1 , _m ) ; // c'esr bk
00671         }
00672         // le résultat du if/else est dans bk, c'est sur la ligne k de _inv, est déjŕ en place
00673 
00674         // Ak+ = Ak+ - d * bk // [k,m] - [k,1] * [1,m] = [k,m]
00675         MatMul( matAux , d , ctPuisb , k , 1 , _m ) ; // matAux(k,m) = d(k) * cPuisb(_m)
00676         MatSub( _inv , _inv , matAux , k , _m ) ;  // et bk est déjŕ en place sur la ligne suivante de _inv
00677 
00678 #if Pr
00679         MatPrint( "_inv" , " %10.6f" , _inv , k + 1 , _m ) ;
00680 #endif
00681 
00682     } // for k
00683 
00684     return 0 ;
00685 } // MatPseudoInv
